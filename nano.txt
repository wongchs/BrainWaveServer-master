#include <SPI.h>
#include <nRF24L01.h>
#include <RF24.h>

RF24 radio(9, 10); // CE, CSN
const byte address[6] = "00001";
const int espPin = 7;

void setup() {
  Serial.begin(9600);
  pinMode(espPin, INPUT);
  radio.begin();
  radio.openWritingPipe(address);
  radio.setPALevel(RF24_PA_MIN);
  radio.stopListening();
}

void loop() {
  if (digitalRead(espPin) == HIGH) {
    const char text[] = "Activate";
    bool sent = radio.write(&text, sizeof(text));
    delay(200);  // Debounce delay
  }
}
